\subsection*{Plotting}
$\Delta G = -RT \ln k$
\\
$k = \exp{\frac{-\Delta G^\circ}{RT}} = \frac{P_{CO_2}}{P_{CO}}$
\\~\\
Let the vertical axis be: $\frac{P_{CO}}{P_{CO}+P_{CO_2}}$
\\~\\
vertical axis in terms of k: $\frac{P_{CO}}{P_{CO}+kP_{CO}} = \frac{1}{1+k}$
\subsection*{equations for phase boundary curves}

For each reaction $i=1\ldots4$:
\[
\Delta G_i^\circ(T)=a_i+b_iT \quad\Rightarrow\quad
K_i(T)=\exp\!\left(\frac{-\Delta G_i^\circ(T)}{RT}\right)
\]
and the plotted ordinate is
\[
\%{\rm CO}_i(T)=100\cdot\frac{1}{1+K_i(T)}.
\]
Use $R=8.314462618\ \mathrm{J\,mol^{-1}\,K^{-1}}$ and the coefficients
(in J/mol, $T$ in K):
\[
\begin{aligned}
&\Delta G_1^\circ=-231.125\times10^3-164.186\,T,\\
&\Delta G_2^\circ=-545.1\times10^3+150.264\,T,\\
&\Delta G_3^\circ=-592.2\times10^3+199.264\,T,\\
&\Delta G_4^\circ=-360.7\times10^3+131.831\,T.
\end{aligned}
\]
