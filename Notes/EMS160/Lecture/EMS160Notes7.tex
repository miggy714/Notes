\section{Mixing standard states}
\begin{center}

$P_A^\circ \equiv \text{the partial pressure of A in its standard state}$
\\
$\Delta G_{A,g}^\circ \equiv \text{molar free energy of gas}$
\\
$\Delta G_{A,s}^\circ \equiv \text{molar free energy of solid}$
\\~\\
Where, 
\\
$\Delta G = \Delta G_{A,s}^\circ - \Delta G_{A,g}^\circ = 0$
\\
$\Delta G_{A,g}^\circ = \Delta G_{A,s}^\circ$
\\
\begin{center}
    \includegraphics[width=200pt]{activitiesnsolutions2}
\end{center}
In this case B is dissolved in A, denoted by: $A_{[B]}$
\begin{itemize}
    \item $\Delta G_{A \rightarrow A[B],g}^\circ \equiv \text{molar free energy to dissolve B in A in the gas phase}$
    \item $\Delta G_{A \rightarrow A[B],s}^\circ \equiv \text{molar free energy to dissolve B in A in the solid phase}$
    \item At equilibrium:
    \begin{enumerate}
        \item $\Delta G_{A \rightarrow A[B],g}^\circ = \Delta G_{A \rightarrow A[B],s}^\circ$
        \item $\Delta G_{A[B],g}-\Delta G_{A,g}^{\circ}=RT \ln{\frac{P_A}{P_A^{\circ}}}$
    \end{enumerate}
\end{itemize}
